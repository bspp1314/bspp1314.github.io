#!/bin/bash 
 
users="test1,test2,test3"
 
array=(${users//,/ })  
res=""
len=${#array[@]} 
echo $res  
 
for (( i=0;i < $len;i++ )) do  
    if (( "$i+1" == $len ));then
    |   res=$res\'${array[i]}\'
    else
    |   res=$res\'${array[i]}\'\,
    fi  
done 
echo $res          


out 
'test1','test2','test3'
